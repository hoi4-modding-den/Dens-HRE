on_actions = {
    on_startup = {
        effect = {
            XHH = {
                activate_advisor = XHH_fdp_liberal_pm
                set_party_name = {
                    ideology = democratic
                    long_name = XHH_democratic_party_seitz
                    name = XHH_democratic_party_seitz
                }
                set_country_leader_ideology = liberalism 
            }

            clear_array = XHH.XHH_integratable_countries
        }
    }
    on_ruling_party_change = {
        effect = {
            if = {
                limit = {
                    original_tag = SHL
                }
                if = {
                    limit = {
                        OR = {
                            has_government = fascism
                            AND = {
                                has_government = neutrality
                                has_country_flag = XHH_senate_removed
                            }
                        }
                    }
                    set_variable = { HRE_national_stance_welfare = 3 }
                }
                if = {
                    limit = {
                        OR = {
                            has_government = communism
                            has_government = democratic
                            AND = {
                                has_government = neutrality
                                NOT = { has_country_flag = XHH_senate_removed }
                            }
                        }
                    }
                    set_variable = { HRE_national_stance_welfare = 1 }
                }
            }
        }
    }

    on_new_term_election = {
		random_events = {
			100 = XHH_internal_events.6
		}
	}
}
